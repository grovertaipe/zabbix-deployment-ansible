==> ./tasks/04-configure.yml <==
---
- name: Detectar si el agente de Zabbix es zabbix-agent2 o zabbix-agentd
  shell: |
    if [[ -n $(command -v dpkg) ]]; then
      agente_instalado=$(dpkg -l | grep zabbix-agent2 | awk '{print $2}')
    elif [[ -n $(command -v rpm) ]]; then
      agente_instalado=$(rpm -qa | grep zabbix-agent2 | awk -F'-' '{print $1"-"$2}')
    fi
    echo "${agente_instalado:-zabbix-agentd}"
  args:
    executable: /bin/bash
  register: agente_instalado

- name: Configurar directorios y configuraciones de Zabbix
  become: true
  block:
    - name: Definir rutas de configuración
      set_fact:
        config_paths:
          agent2:
            dir: "/etc/zabbix/zabbix_agent2.d/"
            conf: "/etc/zabbix/zabbix_agent2.conf"
            plugins: "/etc/zabbix/zabbix_agent2.d/plugins.d/"
            template: "../files/template_agent2.conf"
          agentd:
            dir: "/etc/zabbix/zabbix_agentd.d/"
            conf: "/etc/zabbix/zabbix_agentd.conf"
            plugins: "/etc/zabbix/zabbix_agentd.d/plugins.d/"
            template: "../files/template_agentd.conf"

    - name: Crear directorios de configuración
      file:
        path: "{{ config_paths[agent_type].dir }}"
        state: directory
        mode: "0755"
      vars:
        agent_type: "{{ 'agent2' if agente_instalado.stdout == 'zabbix-agent2' else 'agentd' }}"

    - name: Copiar plantilla de configuración base
      copy:
        src: "{{ config_paths[agent_type].template }}"
        dest: "{{ config_paths[agent_type].conf }}"
        mode: "0644"
      vars:
        agent_type: "{{ 'agent2' if agente_instalado.stdout == 'zabbix-agent2' else 'agentd' }}"

    - name: Configurar Server en plantilla de Zabbix
      lineinfile:
        path: "{{ config_paths[agent_type].conf }}"
        regexp: "^Server="
        line: "Server={{ client_zabbix_proxy_secrets.zabbix_proxy }}"
      vars:
        agent_type: "{{ 'agent2' if agente_instalado.stdout == 'zabbix-agent2' else 'agentd' }}"

    - name: Configurar ServerActive en plantilla de Zabbix
      lineinfile:
        path: "{{ config_paths[agent_type].conf }}"
        regexp: "^ServerActive="
        line: "ServerActive={{ client_zabbix_proxy_secrets.zabbix_proxy | replace(',', ';') }}"
      vars:
        agent_type: "{{ 'agent2' if agente_instalado.stdout == 'zabbix-agent2' else 'agentd' }}"

    - name: Configurar HostMetadata en plantilla de Zabbix
      lineinfile:
        path: "{{ config_paths[agent_type].conf }}"
        regexp: "^HostMetadata="
        line: "HostMetadata={{ client_zabbix_proxy_secrets.host_meta_data }}"
      vars:
        agent_type: "{{ 'agent2' if agente_instalado.stdout == 'zabbix-agent2' else 'agentd' }}"

    - name: Copiar plugins del agente Zabbix
      copy:
        src: "{{ item }}"
        dest: "{{ config_paths[agent_type].plugins }}"
        mode: "0755"
      loop:
        - "../files/top_cpu.sh"
        - "../files/top_memory.sh"
      vars:
        agent_type: "{{ 'agent2' if agente_instalado.stdout == 'zabbix-agent2' else 'agentd' }}"

    - name: Asegurar que el servicio del agente Zabbix esté habilitado
      systemd:
        name: zabbix-agent2
        enabled: true

#!/bin/bash
# Name: monitor_mem_usage.sh
# Description: Script to check top memory consuming process for 2 seconds

# Change the SECS to total seconds to monitor Memory usage.
# UNIT_TIME is the interval in seconds between each sampling

function report_utilisation {
  # Process collected data
  echo
  echo "High memory utilization :"

  cat /tmp/mem_usage.$$ |
    awk '
      { process[$1]+=$2; }
      END{
        for(i in process)
        {
          printf("%-20s %s\n", i, process[i]) ;
        }
      }' | sort -nrk 2 | head -n 5

  # Remove the temporary log file
  rm /tmp/mem_usage.$$
  exit 0
}

trap 'report_utilisation' INT

SECS=2
UNIT_TIME=2

STEPS=$(( SECS / UNIT_TIME ))


# Collect data in temp file
for((i=0;i<STEPS;i++)); do
    ps -eocomm,pmem | egrep -v '(0.0)|(%MEM)' >> /tmp/mem_usage.$$
    sleep $UNIT_TIME
done
